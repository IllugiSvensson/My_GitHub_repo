#Скрипт для переименовывания файлов изображений по шаблону. 

#!/bin/bash
if [ $# -lt 2 ]
then								
	echo Usage:								
	echo $0 newprefix file1 file2 …							 
	exit 1
fi       
							
prefix=$1        
shift        							
count=0     

for file in $*     							
do   
	count=$[count + 1]
	mv -n "${file}" "${prefix}${count}.jpg"
done
exit 0
